/*
 * Complete the 'powerSum' function below.
 *
 * The function is expected to return an INTEGER.
 * The function accepts following parameters:
 *  1. INTEGER x
 *  2. INTEGER n
 */

int powerSum(int x, int m, int n)
{
    int tmp;
    tmp=1;
    for(int i=1;i<=n;i++)
        tmp=tmp*m;
    if(tmp==x)
       return 1;
    if(tmp>x)
        return 0;
    return powerSum(x,m+1,n)+powerSum(x-tmp,m+1,n);
}
